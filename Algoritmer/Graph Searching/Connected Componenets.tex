Connected(G)
	mark s
	s.d = 0
	Q.Enqueue(s)
	while Q.IsEmpty()
		v = Q.Dequeue()
		for each unmarked neighbor u
			mark u
			u.d = v.d + 1
			u.parrent = v
			Q.Enqueue(u)
		
		for each makred neighbor u
			if (u.d mod 2) = (v.d mod 2)
				return false
	return true
\end{lstlisting}
