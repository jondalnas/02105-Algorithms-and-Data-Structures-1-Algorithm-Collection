Double linked liste, hvor hver knude har en key og noget data\\
\textbf{Pseudokode}:
\begin{lstlisting}[frame=single, mathescape=true]
Search(k)
	for all nodes n in L
		if n.key = k
			return n

Insert(x)
	L.Insert(x)

Delete(x)
	L.Delete(x)
\end{lstlisting}
\textbf{Running Time}:
\begin{itemize}
	\item \texttt{SEARCH(k)}: O(n)
	\item \texttt{INSERT(k)}: O(1)
	\item \texttt{DELETE(k)}: O(1)
\end{itemize}
\textbf{Plads}: O(n)
