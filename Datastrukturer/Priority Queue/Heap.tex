Vi kan bruge en heap til at lave en priority queue\\
\begin{lstlisting}[frame=single, mathescape=true]
Max()
	return H[0]

ExtractMax()
	r = H[0]
	H[0] = H[n]
	n = n - 1
	H.BubbleDown(0)
	return r

IncreaseKey(x, k)
	H[x] = k
	H.BubbleUp(x)

Insert(x)
	n = n + 1
	H[x] = k
	H.BubbleUp(x)
\end{lstlisting}
\textbf{Running time}:
\begin{itemize}
	\item \texttt{MAX()}: O(1)
	\item \texttt{EXSTRACTMAX()}: O(log n)
	\item \texttt{INCREASEKEY(x, k)}: O(log n)
	\item \texttt{INSERT(x)}: O(log n)
\end{itemize}
\textbf{Plads}: O(n)
